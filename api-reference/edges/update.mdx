api: "PATCH /v1/projects/{projectId}/edges/{edgeId}"

<ParamField path="edgeId" type="string" required>Edge UUID.</ParamField>

<ParamField body="sourceHandle" type="string">One of `top`, `bottom`, `left`, `right`.</ParamField>
<ParamField body="targetHandle" type="string">One of `top`, `bottom`, `left`, `right`.</ParamField>
<ParamField body="edgeType" type="string">Custom edge type label.</ParamField>
